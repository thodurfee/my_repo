otherbranch=$( awk -F "," "/Project Last Stable Branch/"  "$metafile"  |  cut  -d , -f2)

proj_abv=$( awk -F "," "/Project Abreviation/"  "$metafile"  |  cut  -d , -f2)
newbranch=$( printf '%s\n' "$proj_abv" "$startyear" "$startmonth" "$inccount" | paste -sd "_")
echo $getcount
echo $inccount
echo $newbranch
#
